alter table if exists agent_events enable row level security;

-- 1. Agent Tasks — task queue for agent work
CREATE TABLE IF NOT EXISTS agent_tasks (
    id uuid default gen_random_uuid() primary key,
    status text not null default 'pending' check (status in ('pending', 'running', 'completed', 'failed', 'cancelled')),
    agent text not null,
    assignee text,
    priority integer default 0,
    session_key text,
    result jsonb,
    metadata jsonb,
    created_at timestamptz default now(),
    updated_at timestamptz default now()
);

-- 2. Agent Events — realtime event stream
CREATE TABLE IF NOT EXISTS agent_events (
    id uuid default gen_random_uuid() primary key,
    agent text not null,
    event_type text not null,
    message text,
    task_id uuid references agent_tasks(id) on delete set null,
    metadata jsonb,
    created_at timestamptz default now()
);

-- Enable realtime for agent_events
alter publication supabase_realtime add table agent_events;

-- 3. Agent Status — heartbeat/online tracking
CREATE TABLE IF NOT EXISTS agent_status (
    id uuid default gen_random_uuid() primary key,
    agent_name text not null unique,
    status text not null default 'offline' check (status in ('online', 'offline', 'busy')),
    last_seen timestamptz default now(),
    metadata jsonb,
    updated_at timestamptz default now()
);

-- 4. Cron Schedules — job scheduling tracking
CREATE TABLE IF NOT EXISTS cron_schedules (
    id uuid default gen_random_uuid() primary key,
    job_id text not null unique,
    schedule text not null,
    host text not null,
    status text not null default 'active' check (status in ('active', 'paused', 'disabled')),
    last_run timestamptz,
    next_run timestamptz,
    run_count integer default 0,
    created_at timestamptz default now(),
    updated_at timestamptz default now()
);
